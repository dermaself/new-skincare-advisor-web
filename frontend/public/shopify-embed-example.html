Dermaself Skin Analysis - Shopify Embed Example

How to embed in your Shopify store:

  1. Deploy your Next.js app to a hosting service (Vercel, Netlify, etc.)
  2. Use the embed URL: https://your-domain.com/embed
  3. Add this button to your Shopify theme:
<button onclick="openSkinAnalysis()" class="skin-analysis-btn">
    Start Skin Analysis
</button>

<script>
function openSkinAnalysis() {
    const iframe = document.createElement('iframe');
    iframe.src = 'https://your-domain.com/embed';
    iframe.style.cssText = 'position:fixed;top:0;left:0;width:100%;height:100%;z-index:9999;border:none;';
    document.body.appendChild(iframe);
    
    // Listen for close message
    window.addEventListener('message', function(event) {
        if (event.data.type === 'SKIN_ANALYSIS_CLOSED') {
            document.body.removeChild(iframe);
        }
    });
}
</script>
